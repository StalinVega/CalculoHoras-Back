-- ==========================================================
-- 2. TABLAS DE CATÁLOGOS (MAESTRAS)

CREATE TABLE tipo_ambiente (
    id SERIAL PRIMARY KEY,
    codigo CHAR(1) NOT NULL UNIQUE,
    nombre VARCHAR(20) NOT NULL
);

CREATE TABLE tipo_emision (
    id SERIAL PRIMARY KEY,
    codigo CHAR(1) NOT NULL UNIQUE,
    nombre VARCHAR(20)
);

CREATE TABLE tipo_comprobante (
    id SERIAL PRIMARY KEY,
    codigo_sri VARCHAR(2) NOT NULL UNIQUE,
    nombre VARCHAR(50) NOT NULL,
    activo BOOLEAN DEFAULT TRUE
);

CREATE TABLE tipo_identificacion (
    id SERIAL PRIMARY KEY,
    codigo VARCHAR(2) NOT NULL UNIQUE,
    nombre VARCHAR(50) NOT NULL
);

CREATE TABLE sri_forma_pago (
    id SERIAL PRIMARY KEY,
    codigo_sri VARCHAR(5) NOT NULL UNIQUE,
    nombre VARCHAR(100) NOT NULL
);

-- Tabla de impuestos detallada (mejora del Script 2)
CREATE TABLE impuesto (
    id SERIAL PRIMARY KEY,
    codigo VARCHAR(2) NOT NULL,
    codigo_porcentaje VARCHAR(2) NOT NULL,
    porcentaje NUMERIC(5,2) NOT NULL,
    descripcion TEXT,
    activo BOOLEAN DEFAULT TRUE,
    UNIQUE(codigo, codigo_porcentaje)
);

-- Configuración SRI por ambiente (mejora del Script 2)
CREATE TABLE sri_config (
    ambiente_id INTEGER PRIMARY KEY REFERENCES tipo_ambiente(id),
    url_recepcion TEXT NOT NULL,
    url_autorizacion TEXT NOT NULL,
    url_recepcion_masiva TEXT,
    version VARCHAR(10) DEFAULT '1.1.0'
);

-- ==========================================================
-- 3. ESTRUCTURA DE LA EMPRESA
-- ==========================================================

CREATE TABLE empresa (
    id SERIAL PRIMARY KEY,
    ruc VARCHAR(13) UNIQUE NOT NULL,
    razon_social VARCHAR(255) NOT NULL,
    nombre_comercial VARCHAR(255),
    direccion_matriz TEXT NOT NULL,
    obligado_contabilidad BOOLEAN DEFAULT FALSE,
    agente_retencion VARCHAR(20),
    contribuyente_especial VARCHAR(20),
    p12_ruta TEXT,
    p12_clave TEXT,
    ambiente_default_id INTEGER REFERENCES tipo_ambiente(id) DEFAULT 1,
    creado_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    actualizado_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE establecimiento (
    id SERIAL PRIMARY KEY,
    empresa_id INTEGER NOT NULL REFERENCES empresa(id) ON DELETE CASCADE,
    codigo VARCHAR(3) NOT NULL,
    nombre_comercial TEXT,
    direccion TEXT NOT NULL,
    ambiente_id INTEGER REFERENCES tipo_ambiente(id),
    usa_ambiente_emisor BOOLEAN DEFAULT TRUE,
    activo BOOLEAN DEFAULT TRUE,
    creado_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(empresa_id, codigo)
);

CREATE TABLE punto_emision (
    id SERIAL PRIMARY KEY,
    establecimiento_id INTEGER NOT NULL REFERENCES establecimiento(id) ON DELETE CASCADE,
    codigo VARCHAR(3) NOT NULL,
    activo BOOLEAN DEFAULT TRUE,
    creado_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(establecimiento_id, codigo)
);

CREATE TABLE secuencial_comprobante (
    id SERIAL PRIMARY KEY,
    punto_emision_id INTEGER NOT NULL REFERENCES punto_emision(id) ON DELETE CASCADE,
    tipo_comprobante_id INTEGER NOT NULL REFERENCES tipo_comprobante(id),
    ultimo_valor INTEGER DEFAULT 0,
    UNIQUE(punto_emision_id, tipo_comprobante_id)
);

-- ==========================================================
-- 4. ENTIDADES COMERCIALES
-- ==========================================================

CREATE TABLE cliente (
    id SERIAL PRIMARY KEY,
    empresa_id INTEGER NOT NULL REFERENCES empresa(id) ON DELETE CASCADE,
    tipo_identificacion_id INTEGER REFERENCES tipo_identificacion(id),
    identificacion VARCHAR(20) NOT NULL,
    razon_social VARCHAR(255) NOT NULL,
    direccion TEXT,
    email VARCHAR(255) NOT NULL,
    telefono VARCHAR(20),
    activo BOOLEAN DEFAULT TRUE,
    creado_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    actualizado_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(empresa_id, identificacion) -- LA CLAVE DEL SAAS
);

CREATE TABLE producto (
    id SERIAL PRIMARY KEY,
    empresa_id INTEGER REFERENCES empresa(id) ON DELETE CASCADE,
    codigo_principal VARCHAR(50) NOT NULL,
    codigo_auxiliar VARCHAR(50),
    nombre TEXT NOT NULL,
    precio_unitario NUMERIC(18,6) NOT NULL,
    codigo_impuesto_sri VARCHAR(5) DEFAULT '2', -- IVA
    codigo_porcentaje_sri VARCHAR(5) DEFAULT '2', -- 12% o 15%
    tarifa_iva NUMERIC(5,2) DEFAULT 15.00,
    ice_porcentaje NUMERIC(5,2) DEFAULT 0,
    activo BOOLEAN DEFAULT TRUE,
    UNIQUE(empresa_id, codigo_principal)
);

-- ==========================================================
-- 5. TRANSACCIONAL (CORE DE FACTURACIÓN)
-- ==========================================================

CREATE TABLE comprobante (
    id BIGSERIAL PRIMARY KEY,
    empresa_id INTEGER NOT NULL REFERENCES empresa(id),
    punto_emision_id INTEGER NOT NULL REFERENCES punto_emision(id),
    cliente_id INTEGER NOT NULL REFERENCES cliente(id),
    tipo_comprobante_id INTEGER NOT NULL REFERENCES tipo_comprobante(id),
    
    ambiente_id INTEGER NOT NULL REFERENCES tipo_ambiente(id),
    emision_id INTEGER NOT NULL REFERENCES tipo_emision(id),
    establecimiento_cod_snapshot VARCHAR(3) NOT NULL,
    punto_emision_cod_snapshot VARCHAR(3) NOT NULL,
    
    secuencial VARCHAR(9) NOT NULL,
    clave_acceso VARCHAR(49) UNIQUE,
    fecha_emision TIMESTAMP NOT NULL,
    
    total_sin_impuestos NUMERIC(14,2) NOT NULL,
    total_descuento NUMERIC(14,2) DEFAULT 0,
    total_iva NUMERIC(14,2) NOT NULL,
    total_ice NUMERIC(14,2) DEFAULT 0,
    importe_total NUMERIC(14,2) NOT NULL,
    
    estado VARCHAR(20) NOT NULL DEFAULT 'CREADO',
    estado_sri VARCHAR(20) DEFAULT 'PENDIENTE',
    
    origen_sistema VARCHAR(50),
    origen_id VARCHAR(100),
    
    numero_autorizacion VARCHAR(49),
    fecha_autorizacion TIMESTAMP,
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(origen_sistema, origen_id)
);

CREATE TABLE comprobante_detalle (

-- Función para tablas con "updated_at" (comprobante)
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Función para tablas con "actualizado_at" (empresa, cliente)
CREATE OR REPLACE FUNCTION update_actualizado_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.actualizado_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- GENERAMOS LA CLAVE DE ACCESO Y EL DIGITO DIGITADOR CON FUNCIONE SY trigger
-- ==========================================================
-- 13. AUTOMATIZACIÓN DE CLAVE DE ACCESO (SRI ECUADOR)
-- ==========================================================

-- 1. Función para calcular el Dígito Verificador (Módulo 11)
-- Requerido por el SRI para el último dígito de la clave de acceso
CREATE OR REPLACE FUNCTION calcular_digito_verificador(clave_48 TEXT)
RETURNS CHAR(1) AS $$
DECLARE
    suma INTEGER := 0;
    factor INTEGER := 2;
    i INTEGER;
    digito INTEGER;
BEGIN
    FOR i IN REVERSE length(clave_48)..1 LOOP
        suma := suma + (CAST(substring(clave_48 FROM i FOR 1) AS INTEGER) * factor);
        factor := factor + 1;
        IF factor > 7 THEN factor := 2; END IF;
    END LOOP;
    
    digito := 11 - (suma % 11);
    
    IF digito = 11 THEN RETURN '0';
    ELSIF digito = 10 THEN RETURN '1';
    ELSE RETURN CAST(digito AS CHAR(1));
    END IF;
END;
$$ LANGUAGE plpgsql;

-- 2. Función que arma la estructura de la clave de 49 dígitos
CREATE OR REPLACE FUNCTION generar_clave_acceso_sri(
    fecha TIMESTAMP,
    tipo_comp VARCHAR,
    ruc_empresa VARCHAR,
    ambiente VARCHAR,
    serie VARCHAR,
    secuencial VARCHAR,
    tipo_emision VARCHAR
) RETURNS VARCHAR(49) AS $$
DECLARE
    clave_48 VARCHAR(48);
    codigo_num VARCHAR(8);
BEGIN
    -- El código numérico por estándar suele ser el mismo secuencial
    codigo_num := LPAD(secuencial, 8, '0'); 
    
    clave_48 := to_char(fecha, 'DDMMYYYY') || 
                tipo_comp || 
                ruc_empresa || 
                ambiente || 
                serie || 
                LPAD(secuencial, 9, '0') || 
                codigo_num || 
                tipo_emision;
                
    RETURN clave_48 || calcular_digito_verificador(clave_48);
END;
$$ LANGUAGE plpgsql;

-- 3. Trigger Function: El "Vigilante" que une todo antes de guardar
CREATE OR REPLACE FUNCTION trg_logica_antes_insertar_comprobante()
RETURNS TRIGGER AS $$
DECLARE
    v_ruc VARCHAR;
    v_tipo_comp VARCHAR;
BEGIN
    -- 1. Obtener datos necesarios de las tablas maestras
    SELECT ruc INTO v_ruc FROM empresa WHERE id = NEW.empresa_id;
    SELECT codigo_sri INTO v_tipo_comp FROM tipo_comprobante WHERE id = NEW.tipo_comprobante_id;

    -- 2. Generar la Clave de Acceso automáticamente
    NEW.clave_acceso := generar_clave_acceso_sri(
        NEW.fecha_emision,
        v_tipo_comp,
        v_ruc,
        CAST(NEW.ambiente_id AS VARCHAR),
        NEW.establecimiento_cod_snapshot || NEW.punto_emision_cod_snapshot,
        NEW.secuencial,
        CAST(NEW.emision_id AS VARCHAR)
    );
    
    -- 3. Otras lógicas automáticas (opcional)
    -- Por ejemplo, asegurar que el estado inicial sea siempre CREADO
    IF NEW.estado IS NULL THEN
        NEW.estado := 'CREADO';
    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 4. Crear el Trigger en la tabla comprobante
DROP TRIGGER IF EXISTS trg_comprobante_auto_clave ON comprobante;
CREATE TRIGGER trg_comprobante_auto_clave
    BEFORE INSERT ON comprobante
    FOR EACH ROW
    EXECUTE FUNCTION trg_logica_antes_insertar_comprobante();
